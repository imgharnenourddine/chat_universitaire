3)C'est quoi un Chunk ?

	Chunk = morceau de texte

	PDF original (10 pages) ->LangChain4j découpe en chunks

4)C'est quoi un Vecteur ?
	Vecteur = liste de nombres qui représente le sens d'un texte
	Chunk 1 : "Les examens du S1 sont en janvier"
	OpenAI convertit en vecteur :
	[0.23, 0.45, 0.12, 0.67, 0.34, 0.89...]
	(liste de 1536 nombres)

5)Pourquoi des vecteurs ?
	Question étudiant :
"Quand sont les examens ?"
        ↓
Convertie en vecteur :
[0.24, 0.44, 0.13, 0.66, 0.35, 0.88...]
        ↓
Qdrant compare ce vecteur
avec tous les vecteurs stockés
        ↓
Chunk 1 est le plus proche ! ✅
→ "Les examens du S1 sont en janvier"

    ├── Lit le PDF

8)C'est quoi le Pipeline RAG ?

	RAG = Retrieval Augmented Generation=Génération Augmentée par Récupération

	Pipeline = suite d'étapes dans l'ordre

	Pipeline RAG = 5 étapes :

	Étape 1 — INGESTION (admin uploade PDF)
	PDF → chunks → vecteurs → Qdrant ✅

	Étape 2 — QUESTION (étudiant pose question)
	"Quand sont les examens ?"

	Étape 3 — RETRIEVAL (récupération)
	Question → vecteur → Qdrant cherche
	→ trouve chunks similaires ✅

	Étape 4 — AUGMENTATION
	Question + chunks trouvés = prompt complet

	Étape 5 — GENERATION
	OpenAI génère réponse basée sur les chunks ✅

9) C'est quoi l'Indexation ?

	Indexation = préparer un document pour la recherche

	Admin uploade "calendrier_examens.pdf"
        ↓
	statut = EN_ATTENTE

	INDEXATION commence :
	    1. Lire le PDF ✅
	    2. Découper en chunks ✅
	    3. Vectoriser avec OpenAI ✅
	    4. Stocker vecteurs dans Qdrant ✅
        ↓
	statut = INDEXE ✅

@Configuration
public class LangChainConfig {

    @Value("${langchain4j.mistral-ai.api-key}")
    private String mistralApiKey;

    @Value("${qdrant.host}")
    private String qdrantHost;

    @Value("${qdrant.port}")

                .apiKey(mistralApiKey)#Commence à construire le modèle de chat
                .modelName("mistral-small-latest")#Clé API pour accéder à Mistral AI 
                .build();
    }

    // Modèle d'embeddings (vectorisation)
    @Bean
    public MistralAiEmbeddingModel embeddingModel() {
        return MistralAiEmbeddingModel.builder()
                .apiKey(mistralApiKey)
                .modelName("mistral-embed")
                .build();
    }

    // Base de données vectorielle (Qdrant)
    @Bean
    public EmbeddingStore<TextSegment> embeddingStore() {Commence à construire la connexion Qdrant	
        return QdrantEmbeddingStore.builder()#STOCKER les vecteurs (lors de l'indexation)/CHERCHER les vecteurs (lors d'une question)
                .host(qdrantHost)
                .port(qdrantPort)
                .collectionName(collectionName)
                .build();
    }
}

package ma.enset.ask_enset.service;

import dev.langchain4j.data.document.Document;
import dev.langchain4j.data.document.DocumentSplitter;
import dev.langchain4j.data.document.splitter.DocumentSplitters;
import dev.langchain4j.data.embedding.Embedding;
import dev.langchain4j.data.segment.TextSegment;
import dev.langchain4j.model.mistralai.MistralAiEmbeddingModel;
import dev.langchain4j.store.embedding.EmbeddingStore;
import io.minio.MinioClient;
import io.minio.PutObjectArgs;
import io.minio.GetObjectArgs;
import lombok.RequiredArgsConstructor;
import ma.enset.ask_enset.model.Document_enset;
import ma.enset.ask_enset.model.User;
import ma.enset.ask_enset.repository.DocumentRepository;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.stereotype.Service;
import org.springframework.web.multipart.MultipartFile;

import java.io.InputStream;
import java.util.List;

@Service  #Spring crée cette classe ✅
@RequiredArgsConstructor
public class DocumentService {

    private final MinioClient minioClient;#Pour uploader/récupérer les PDFs dans MinIO
    private final MistralAiEmbeddingModel embeddingModel;#Pour convertir les chunks en vecteurs
    private final EmbeddingStore<TextSegment> embeddingStore;#Pour stocker les vecteurs dans Qdrant
    private final DocumentRepository documentRepository;#Pour sauvegarder les infos du document dans PostgreSQL

    @Value("${minio.bucket-name}")#bucketName = "enset-documents" ✅
    private String bucketName;

    // Upload et indexation du PDF
    public Document_enset uploadDocument(
            MultipartFile file,#le fichier PDF uploadé
            String description,#"Calendrier des examens S1"
            String categorie,#"EXAMEN"
            User uploadedBy) throws Exception {# l'admin qui uploade

        String fileName = file.getOriginalFilename();#Récupère le nom du fichier
        String cheminMinio = categorie.toLowerCase() + "/" + fileName;

        // 1. Sauvegarder dans MinIO
        minioClient.putObject(#C'est la commande qui uploade le fichier dans MinIO
                PutObjectArgs.builder()
                        .bucket(bucketName)
                        .object(cheminMinio)
                        .stream(file.getInputStream(),#// Ouvre le fichier PDF pour lecture
                                file.getSize(), -1)# // taille du fichier avec -1 MinIO découpe en parties si nécessaire
                        .contentType(file.getContentType())#"application/pdf"
                        .build()#Fichier stocké dans MinIO
        );

        // 2. Sauvegarder dans PostgreSQL
        Document_enset document = new Document_enset();
        document.setNomFichier(fileName);
        document.setDescription(description);
        document.setCategorie(
                Document_enset.Categorie.valueOf(categorie)
        );
        document.setCheminMinio(cheminMinio);
        document.setStatut(Document_enset.Statut.EN_ATTENTE);
        document.setUploadedBy(uploadedBy);
        documentRepository.save(document);

        // 3. Indexer dans Qdrant
        indexDocument(document, file);#C'est un **appel de méthode** — on appelle la méthode privée `indexDocument()` qu'on a définie plus bas dans la même classe.

        return document;
    }

    // Indexation dans Qdrant
    private void indexDocument(
            Document_enset document,
            MultipartFile file) throws Exception {

        // Lire le contenu du PDF
        String content = new String(file.getBytes());#lit tous les bytes du fichier convertit en texte lisible

        // Créer un document LangChain4j
        Document doc = Document.from(content);#Document LangChain4j contenant le texte ✅ LangChain4j travaille avec son propre type Document

        // Découper en chunks
        DocumentSplitter splitter = DocumentSplitters
                .recursive(500, 50);#découpe intelligemment essaie de couper aux paragraphes → puis aux phrases→ puis aux mots→ jamais au milieu d'un mot ✅500 = taille max d'un chunken caractères 50 = chevauchement (overlap)
        List<TextSegment> chunks = splitter.split(doc);#TextSegment = un chunk de texte LangChain4j,List<TextSegment> = liste de chunks ✅

        // Vectoriser et stocker dans Qdrant
        for (TextSegment chunk : chunks) {#Pour chaque chunk dans la liste :→ vectorise→ stocke dans Qdrant
            Embedding embedding = embeddingModel
                    .embed(chunk.text())#chunk.text()->Récupère le texte du chunk  ->Mistral convertit en vecteur :[0.23, 0.45, 0.12, 0.67... 1024 nombres]
                    .content();extrait juste le vecteur dans  EmbeddingResponse returner par embed.
            embeddingStore.add(embedding, chunk);#envoie à Qdrant: Qdrant stocke les deux ensemble ,Vecteur → pour la recherche rapide ,Texte   → pour retourner le contenu
        }

        // Mettre à jour le statut
        document.setStatut(Document_enset.Statut.INDEXE);
        documentRepository.save(document);#Hibernate voit que id=1 existe déjà
        ↓
Génère UPDATE au lieu de INSERT :
    }

    // Liste tous les documents
    public List<Document_enset> getAllDocuments() {
        return documentRepository.findAll();utile en ca ou Admin ouvre son dashboard
        
    }
}

package ma.enset.ask_enset.controller;

import lombok.RequiredArgsConstructor;
import ma.enset.ask_enset.model.Document_enset;
import ma.enset.ask_enset.model.User;
import ma.enset.ask_enset.repository.UserRepository;
import ma.enset.ask_enset.service.DocumentService;
import org.springframework.http.ResponseEntity;
import org.springframework.security.core.Authentication;
import org.springframework.web.bind.annotation.*;
import org.springframework.web.multipart.MultipartFile;

import java.util.List;

@RestController
@RequestMapping("/api/documents")
@RequiredArgsConstructor
@CrossOrigin(origins = "http://localhost:3000")
public class DocumentController {

    private final DocumentService documentService;
    private final UserRepository userRepository;

    // Upload PDF
    @PostMapping("/upload")
    public ResponseEntity<Document_enset> upload(
            @RequestParam("file") MultipartFile file,
            @RequestParam("description") String description,
            @RequestParam("categorie") String categorie,
            Authentication authentication) {
        try {
            // Récupère l'admin connecté
            String email = authentication.getName();
            User admin = userRepository
                    .findByEmail(email)
                    .orElseThrow();

            // Upload et indexation
            Document_enset document = documentService
                    .uploadDocument(
                            file,
                            description,
                            categorie,
                            admin
                    );

            return ResponseEntity.ok(document);

        } catch (Exception e) {
            return ResponseEntity
                    .internalServerError()
                    .build();
        }
    }

    // Liste tous les documents
    @GetMapping
    public ResponseEntity<List<Document_enset>> getAllDocuments() {
        return ResponseEntity.ok(
                documentService.getAllDocuments()
        );
    }
}
