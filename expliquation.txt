------------------------------------------------------------------comparaison entre flask et spring boot-------------------------------------------------------------------------

1)Avec Flask (ce que tu connais)
	1. Tu crées un dossier manuellement
	2. Tu crées app.py manuellement
	3. Tu installes les libs avec pip
	4. Tu écris tout from scratch

2)Avec Spring Boot (nouveau)
	1. Tu vas sur start.spring.io
	2. Tu remplis un formulaire
	3. Tu cliques Generate
	4. Il te donne un projet complet prêt à l'emploi

	💡 Spring Boot génère automatiquement toute la structure du projet — tu n'as rien à créer manuellement !


3)C'est quoi start.spring.io ?

	C'est le générateur officiel de projets Spring Boot — comme un formulaire IKEA.
		Tu remplis ce que tu veux
      			  ↓
		Il génère un fichier .zip
      			  ↓
		Tu extrais → projet prêt ✅

4)Maven  → utilise pom.xml    (comme pip avec requirements.txt)

5)Packaging — Jar ou War
	Jar → application autonome ✅ (comme un .exe)
	War → nécessite un serveur externe (ancien)
	
	On choisit Jar — plus simple, Spring Boot a son propre serveur intégré

6)Avec JPA/Hibernate tu écris juste ça ?
	
	JPA = Java Persistence API
		C'est une interface standard qui dit comment Java doit communiquer avec une base de données.

	// AVEC JPA — une seule ligne ✅
		userRepository.save(user);
```

		> 💡 Hibernate génère automatiquement le SQL pour toi !

---

		## La différence JPA vs Hibernate
```
		JPA       = le PATRON qui donne les ordres
		Hibernate = l'EMPLOYÉ qui exécute les ordres
```

	Plus concrètement :
```
		JPA dit :
		"Il faut une méthode save() pour sauvegarder"
		"Il faut une méthode findAll() pour récupérer"
		"Il faut une méthode delete() pour supprimer"
              	↓
		Hibernate dit :
		"OK je vais créer ces méthodes"
		"Pour save() je vais écrire INSERT INTO..."
		"Pour findAll() je vais écrire SELECT * FROM..."
		"Pour delete() je vais écrire DELETE FROM..."

1)package ma.enset.ask_enset.model;



import jakarta.persistence.*;//c'est le package JPA
import lombok.*;//c'est le package Lombok
import java.time.LocalDateTime;

@Entity//Dit à Hibernate "cette classe est une table" dans la base de données
@Table(name = "users")//Dit à Hibernate que le nom de la table dans PostgreSQL sera users
@Getter//Génère automatiquement tous les getters :
@Setter//Génère automatiquement tous les setters :
@NoArgsConstructor//Génère un constructeur vide automatiquement :
@AllArgsConstructor//Génère un constructeur avec tous les attributs :
public class User {

    @Id//Dit que cet attribut est la clé primaire de la table
    @GeneratedValue(strategy = GenerationType.IDENTITY)//L'id est généré automatiquement par PostgreSQL (1, 2, 3...)
    private Long id;

    @Column(nullable = false)
    private String nom;

    @Column(nullable = false)
    private String prenom;

    @Column(nullable = false, unique = true)
    private String email;

    @Column(nullable = false)
    private String password;
	
    @OneToMany(mappedBy = "user", cascade = CascadeType.ALL)/Une user → contient → Plusieurs conversation //user supprimée->Tous les conversation supprimés automatiquement ✅
    private List<Conversation> conversations;//si en faire user.getConversations() il nous donne les conversation

    @Enumerated(EnumType.STRING)/Ça dit à Hibernate comment stocker l'enum dans PostgreSQL ,stocke "ETUDIANT" ou "ADMIN" 
    @Column(nullable = false)
    private Role role;/`Role` → c'est le **type de données** (comme String, Integer...)

    @Column(name = "image_path")//le nom de la colonne dans PostgreSQL sera image_path
    private String imagePath = "profiles/default.jpg";

    @Column(name = "created_at")
    private LocalDateTime createdAt;

    @Column(name = "updated_at")
    private LocalDateTime updatedAt;

    @PrePersist//exécuté **automatiquement** juste avant la création d'un utilisateur
    protected void onCreate() {
        createdAt = LocalDateTime.now();
        updatedAt = LocalDateTime.now();
    }

    @PreUpdate//exécuté **automatiquement** quand tu modifies un utilisateur
    protected void onUpdate() {
        updatedAt = LocalDateTime.now();
    }

    public enum Role {/ enum type spécial (liste de valeurs fixes),Role     → nom du type
        ETUDIANT, ADMIN
    }
}

1)package ma.enset.ask_enset.model;



import jakarta.persistence.*;
import lombok.*;
import java.time.LocalDateTime;
import java.util.List;

@Entity
@Table(name = "conversations")
@Getter
@Setter
@NoArgsConstructor
@AllArgsConstructor
public class Conversation {

    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;

    @ManyToOne //Plusieurs conversations → appartiennent à → Un seul user
    @JoinColumn(name = "user_id", nullable = false)//@JoinColumn → crée la colonne user_id dans la table 
    private User user;//private User user → l'objet User complet (pas juste l'id) conversation.getUser().getNom()  // → "Alami"

    @Column(nullable = false)
    private String titre;

    @OneToMany(mappedBy = "conversation", cascade = CascadeType.ALL)//Une conversation → contient → Plusieurs messages ;//mappedBy = "conversation" → dit à JPA que la relation est déjà définie dans Message.java :
    private List<Message> messages;

    @Column(name = "created_at")
    private LocalDateTime createdAt;

    @Column(name = "updated_at")
    private LocalDateTime updatedAt;

    @PrePersist
    protected void onCreate() {
        createdAt = LocalDateTime.now();
        updatedAt = LocalDateTime.now();
    }

    @PreUpdate
    protected void onUpdate() {
        updatedAt = LocalDateTime.now();
    }
}

1)package ma.enset.ask_enset.model;



import jakarta.persistence.*;
import lombok.*;
import java.time.LocalDateTime;

@Entity
@Table(name = "messages")
@Getter
@Setter
@NoArgsConstructor
@AllArgsConstructor
public class Message {

    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;

    @ManyToOne
    @JoinColumn(name = "conversation_id", nullable = false)
    private Conversation conversation;

    @Column(nullable = false, columnDefinition = "TEXT")//VARCHAR → limité à 255 caractères;TEXT → illimité ✅ — parfait pour les réponses du chatbot qui peuvent être longues
    private String contenu;

    @Enumerated(EnumType.STRING)
    @Column(nullable = false)
    private TypeMessage type;

    @Column(name = "created_at")
    private LocalDateTime createdAt;

    @PrePersist
    protected void onCreate() {
        createdAt = LocalDateTime.now();
    }

    public enum TypeMessage {
        USER, BOT
    }
}

1)package ma.enset.askEnset.model;

import jakarta.persistence.*;
import lombok.*;
import java.time.LocalDateTime;

@Entity
@Table(name = "documents")
@Getter
@Setter
@NoArgsConstructor
@AllArgsConstructor
public class Document {

    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;

    @Column(name = "nom_fichier", nullable = false)
    private String nomFichier;

    @Column(nullable = false)
    private String description;

    @Enumerated(EnumType.STRING)
    @Column(nullable = false)
    private Categorie categorie;

    @Column(name = "chemin_minio", nullable = false)
    private String cheminMinio;

    @Enumerated(EnumType.STRING)
    @Column(nullable = false)
    private Statut statut = Statut.EN_ATTENTE;//Quand un PDF est uploadé → statut automatiquement `EN_ATTENTE` :

    @ManyToOne
    @JoinColumn(name = "uploaded_by", nullable = false)
    private User uploadedBy;

    @Column(name = "created_at")
    private LocalDateTime createdAt;

    @PrePersist
    protected void onCreate() {
        createdAt = LocalDateTime.now();
    }

    public enum Categorie {
        EMPLOI_DU_TEMPS, EXAMEN, REGLEMENT, STAGE
    }

    public enum Statut {
        EN_ATTENTE, INDEXE, ERREUR
    }
}

2)package ma.enset.askEnset.security;

import io.jsonwebtoken.Claims; #Claims = les données stockées dans le token (email, expiration...)
import io.jsonwebtoken.Jwts; #Jwts = la classe principale pour créer et lire les tokens JWT
import io.jsonwebtoken.security.Keys;#Keys = classe pour créer la clé de signature
import org.springframework.beans.factory.annotation.Value;#@Value = pour lire les valeurs de application.properties
import org.springframework.stereotype.Service;#@Service = dit à Spring que c'est un service métier

import java.security.Key;#Key = type de la clé de signature

import java.util.Date;

@Service #@Service → Spring Boot crée cette classe automatiquement au démarrage
public class JwtService {

    @Value("${jwt.secret}")
    private String secret;

    @Value("${jwt.expiration}")
    private Long expiration;

    // Générer un token
    public String generateToken(String email) {#Reçoit un email → retourne un token String
        return Jwts.builder() #Commence à construire le token
                .subject(email) #Met l'email dans le token
                .issuedAt(new Date()) #Met la **date de création** du token :
                .expiration(new Date(System.currentTimeMillis() + expiration))#System.currentTimeMillis()` = maintenant en millisecondes
                .signWith(getKey())#**Signe** le token avec la clé secrète :
                .compact();#**Finalise** et retourne le token comme String :
    }

    // Extraire l'email depuis le token
    public String extractEmail(String token) {
        return getClaims(token).getSubject();#`getClaims(token)` → lit les données du token/`.getSubject()` → récupère l'email qu'on a mis avec `.subject(email)`
    }

    // Vérifier si le token est valide
    public boolean isTokenValid(String token) {
        try {
            return !getClaims(token) #`getClaims(token).getExpiration()` → récupère la date d'expiration
                    .getExpiration()
                    .before(new Date());#`.before(new Date())` → est-ce que la date est **avant** maintenant ?
        } catch (Exception e) {
            return false;
        }
    }

    // Méthodes privées
    private Claims getClaims(String token) {
        return Jwts.parser() #`Jwts.parser()` → commence à **lire** le toke
                .verifyWith((javax.crypto.SecretKey) getKey()) #.verifyWith(getKey())` → vérifie la signature avec la clé secrète
                .build()
                .parseSignedClaims(token)#parseSignedClaims(token)` → décode le token
                .getPayload();#getPayload()` → retourne les données (email, expiration...)
    }

    private Key getKey() {
        return Keys.hmacShaKeyFor(secret.getBytes()); #`secret.getBytes()` → convertit le texte secret en bytes #crée une clé cryptographique ->"askEnsetSecretKey..." → bytes → clé cryptographique 🔑
    }
}
